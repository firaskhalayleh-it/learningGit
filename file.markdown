``` bash
# Initialize a new Git repository named gitTreaning
git init gitTreaning

# Navigate into the repository directory
cd gitTreaning/

# Clone an existing remote repository into the local repository (note the correct command syntax)
git clone https://github.com/firaskhalayleh-it/learningGit

vim file.markdown

# Stage the changes to prepare for the commit
git add file.markdown

# Commit the changes with a commit message
git commit -m "training with git"

# Push the committed changes to the remote repository (assuming you have already set up the remote repository correctly)
git push origin main
```
